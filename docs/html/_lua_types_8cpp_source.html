175  do

178  keyPath.pop();

180

183  else if (leafIt != currTable->leafSet.end())

223  Table::KeyPath Table::tokenizePath(const std::string &searchPath) const
224  {
225  KeyPath keyPath;
226  if (searchPath.size() == 0)
227  return keyPath;
228  string::const_iterator currIndex = searchPath.begin();
229  if (!isToken(currIndex))
230  throw path_lookup_exception("Invalid starting token character");
231 
232  while (currIndex != searchPath.end())
233  {
234  // the index of the first character of the field
235  string::const_iterator startIndex = currIndex + 1;
236  char token = *currIndex;
237  //iterate to next token. If we are at the end then this'd be
238  // the last iteration
239  while (++currIndex != searchPath.end() && !isToken(currIndex));
240 
241  string field(startIndex, currIndex);
242  switch (token)
243  {
244  case STRING_TOKEN:
245  keyPath.emplace(Key(field));
246  break;
247  case NUMBER_TOKEN:
248  keyPath.emplace(Key(Key::Type::NUMBER,field));
249  break;
250  }
251  }
252 
253  return keyPath;
254  }
